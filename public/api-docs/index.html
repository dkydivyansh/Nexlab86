Code Examples

JavaScript


# GET Request
async function getData(id, accessKey) {
    const response = await fetch(
        `/api/v1/data.php?id=${id}&access_key=${accessKey}`
    );
    return await response.json();
}

# PUT Request
async function updateData(id, accessKey, type, value) {
    const response = await fetch(
        `/api/v1/data.php?id=${id}`,
        {
            method: 'PUT',
            headers: {
                'Content-Type': 'application/json'
            },
            body: JSON.stringify({
                access_key: accessKey,
                type: type,
                value: value
            })
        }
    );
    return await response.json();
}

Python


import requests
import json

# GET Request
def get_data(id, access_key):
    response = requests.get(
        '/api/v1/data.php',
        params={'id': id, 'access_key': access_key}
    )
    return response.json()

# PUT Request
def update_data(id, access_key, type, value):
    response = requests.put(
        f'/api/v1/data.php?id={id}',
        json={
            'access_key': access_key,
            'type': type,
            'value': value
        }
    )
    return response.json()
